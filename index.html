Quick Start

Up in 30 seconds

terminal 
# Clone
git clone https://github.com/MrLiPx/html-previewer.git
cd html-previewer

# Open directly — no server needed
open app.html      # macOS
start app.html     # Windows
xdg-open app.html # Linux

# Deploy to Netlify / Cloudflare Pages — _redirects is included
# Deploy to Vercel — vercel.json is included
# Deploy to GitHub Pages — Settings → Pages → Deploy from / branch
